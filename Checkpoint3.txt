    const arr = [];

 function parity (arr) {
    
    let part = "";

    if (arr.length >= 3){
      if ((arr[0]%2) == 0) {
        if ((arr[1]%2) == 0) { part = "even"}
        else if ((arr[2]%2) == 0) { part = "even"}
        else { part = "odd" }

      }else if ((arr[1]%2) == 0) {
          if (((arr[2]%2) == 0)) { part = "even"}

   }
